\begin{enumerate}
 \item   $f(x)= x^{38} + 4 \ln(x) -3e^{x} + \pi^{3000}$.
 \vspace{3cm}

 
 \item $f(x) =x^5e^{x}\tan(x)$.
 
 \vspace{5cm}
 
 \item $f(x) = \cot(x)$. (Write $\cot(x)=\dfrac{\cos(x)}{\sin(x)}$ and use the quotient rule.)
 
 \vspace{8cm}
 
 \item  $f(x) = \cos(e^{x}+ 13x^{11}-\sin(x))$. 
 

\end{enumerate}
