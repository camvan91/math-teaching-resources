 \item Use Cramer's rule to find the value of $x$ in the solution to the following system of equations:
\[
 \begin{array}{ccccccc}
  x&+&5y&-&3z&=&2\\
 2x&+&2y&+&4z&=&-1\\
-2x&-&7y&+&3z&=&0
 \end{array}

\bigskip

 \item The matrix $C$ obtained by switching rows 2 and 3 of $A$.

\bigskip

 \item The matrix $2A$.
\end{enumerate}
